get_value

Gets a value from an object with the option to have a default value. In the future, this should accept dot-notation and possibly array index notation, like so:

get_value(options, 'foo.bar.baz[3]', 42);

function get_value(object, key, default_value) {
  if(key in object) {
    return object[key];
  }
  return default_value;
}

exports.get_value = get_value;
